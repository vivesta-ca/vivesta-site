  <script>

    let allProviders = [];


    function applyFilter(service) {

      const pills = document.querySelectorAll(".pill");

      pills.forEach(p => {

        if (p.dataset.service === service) {

          p.classList.add("active");

        } else if (service === "all" && p.dataset.service === "all") {

          p.classList.add("active");

        } else if (service === "all" && p.dataset.service !== "all") {

          p.classList.remove("active");

        } else {

          p.classList.remove("active");

        }

      });


      if (service === "all") {

        renderProviders(allProviders);

        return;

      }


      const filtered = allProviders.filter(p => {

        const services = (p.Services || "").toLowerCase();

        return services.includes(service.toLowerCase());

      });

      renderProviders(filtered);

    }


    function renderProviders(list) {

      const container = document.getElementById("providers-list");


      if (!list || !list.length) {

        container.innerHTML = `

          <h2 class="empty">No providers found</h2>

          <p class="empty">Try a different filter or check back soon.</p>

        `;

        return;

      }


      const cards = list.map(p => {

        const name = p.Name || "Provider";

        const services = p.Services || "";

        const phone = p.Phone || "";

        const email = p.Email || "";

        const rate = p.Rate || "";

        const area = p.Area || "Gatineau";


        return `

          <article class="provider-card">

            <h3>${name}</h3>

            <p class="meta-line">${rate ? rate + " · " : ""}${area}</p>

            ${services ? `<p class="meta-line">${services}</p>` : ""}

            ${phone ? `<p class="meta-line">Phone: ${phone}</p>` : ""}

            ${

              email

                ? `<a class="contact-btn" href="mailto:${email}?subject=Vivesta inquiry">Contact ${name}</a>`

                : ""

            }

          </article>

        `;

      }).join("");


      container.innerHTML = `

        <h2>${list.length} provider${list.length === 1 ? "" : "s"} found</h2>

        <div class="providers-grid">

          ${cards}

        </div>

      `;

    }


    function loadProviders() {

      fetch("/.netlify/functions/providers")

        .then(res => {

          if (!res.ok) {

            throw new Error("Network error: " + res.status);

          }

          return res.json();

        })

        .then(data => {

          // Expecting an array of Airtable record fields

          allProviders = data || [];

          renderProviders(allProviders);

        })

        .catch(err => {

          console.error(err);

          const container = document.getElementById("providers-list");

          container.innerHTML = `

            <h2 class="error">Error loading providers</h2>

            <p class="error">Please try again later.</p>

          `;

        });

    }


    document.addEventListener("DOMContentLoaded", () => {

      loadProviders();


      document.querySelectorAll(".pill").forEach(pill => {

        pill.addEventListener("click", () => {

          const service = pill.dataset.service;

          applyFilter(service);

        });

      });

    });
